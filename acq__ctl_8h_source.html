74 #ifndef ACQ_CTL_H_
75 #define ACQ_CTL_H_
76 
77 /*
78 * Command codes to the VME processor task.
79 */
80 #define INIT_ACQ 0x11 /* Initialize the run-time acquisition system */
81 #define START_ACQ 0x22 /* Start data acquisition */
82 #define STOP_ACQ 0x33 /* Stop data acquisition */
83 #define STATUS_ACQ 0x44 /* Get acquisition status */
84 #define PAC_FILE 0x55 /* Get PAC file name */
85 #define HOST 0x66 /* Get host Ethernet address */
86 #define ZERO_CLK 0x77 /* Zero the acquisition VME clock */
87 
88 
89 /*
90 * Acquisition status codes
91 */
92 #define ACQ_RUN 1 /* Acquisition running */
93 #define ACQ_STOP 2 /* Acquisition stopped */
94 #define ACQ_UNINIT 3 /* Acquisition not initialized */
95 
96 /*
97 * Returned status codes.
98 */
99 #define ACQ_OK 0 /* All went well */
100 
101 /*
102 * Error status codes
103 */
104 #define ACQ_UNKNOWN_COMMAND (-1)
105 
106 #define ACQ_STP_HALT (-5) /* Acquisition already stopped */
107 #define ACQ_STR_RUN (-4) /* Acquisition already running */
108 #define ACQ_STR_NOINIT (-3) /* Start failed. System has not been
109  initialized. */
110 #define ACQ_INIT_RUN (-6) /* Init error - acquisition is running */
111 #define ACQ_INVALID_TABLE (-2) /* Invalid Acq table index */
112 #define ACQ_CAM_NOEXIST (-10) /* nonexistant crate */
113 #define ACQ_CAM_OFFLINE (-11) /* Crate off-line */
114 #define ACQ_CAM_INHIBIT (-12) /* Crate Inhibit stuck */
115 #define ACQ_CAM_INIT (-13) /* Initialization list CAMAC timeout */
116 #define ACQ_FB_NOEXIST (-20) /* FASTBUS module does not respond */
117 #define ACQ_FB_UNKNOWN (-21) /* Unknown FASTBUS module type */
118 #define ACQ_FERA_UNKNOWN (-30) /* Unknown FERA module type */
119 #define ACQ_FERA_INIT (-31) /* CAMAC timeout initializing FERA module */
120 #define ACQ_NO_KSC2917 (-40) /* No CAMAC interface present */
121 #define ACQ_NO_LRS1131 (-41) /* No VME FASTBUS interface present */
122 #define ACQ_NO_LRS1821 (-42) /* No FASTBUS 1821 sequencer present */
123 #define ACQ_NO_ACROMAG (-43) /* No Acromag digital I/O present */
124 #define ACQ_NO_FERA (-44) /* No FERA interface present */
125 #define ACQ_NO_TRIGGER (-45) /* No ORNL trigger module present */
126 #define ACQ_NO_KSC3982 (-46) /* No List Sequencer present */
127 #define ACQ_SEQ_BUFFERS (-47) /* Sequencer buffers are too small */
128 
129 #define ACQ_COUNT_DWN (-48) /* Count down list too large */
130 #define ACQ_RAW_GATE (-49) /* Raw gate list too large */
131 #define ACQ_CAL_GATE (-50) /* Calculated gate list too large */
132 #define ACQ_CAM_RO (-51) /* Special CAMAC readout list too large */
133 #define ACQ_CAM_UNKNOWN (-52) /* Unknown CAMAC module type */
134 #define ACQ_GATE_MODTYPE (-53) /* Illegal module type in gate spec. */
135 
136 #define ACQ_CR_COND (-54) /* Illegal crate in conditional list */
137 #define ACQ_CR_UNCOND (-55) /* Illegal crate in unconditional list */
138 #define ACQ_CR_LATCH (-56) /* Illegal crate in latch list */
139 #define ACQ_CR_WIND (-57) /* Illegal crate in windup list */
140 #define ACQ_CR_GATE (-58) /* Illegal crate in gate read list */
141 #define ACQ_CR_CAMAC (-59) /* Illegal crate in CAMAC special modu */
142 #define ACQ_CR_RUN (-60) /* Illegal crate in $run list */
143 
144 #define ACQ_NO_AUX (-61) /* No ORNL AUX found */
145 #define ACQ_NO_XIA (-62) /* No XIA modules found */
146 #define ACQ_NO_CAEN_ADC (-63) /* CAEN ADC not present in VME crate */
147 #define ACQ_NO_CAEN_TDC (-64) /* CAEN ADC not present in VME crate */
